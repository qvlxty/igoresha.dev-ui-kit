# Пример компонента Модали


```tsx
function ModalSandbox(args) {
    const [visible, setVisible] = React.useState(false)

    return (
      <>
        <Button onClick={() => setVisible(true)} >Modal open</Button>
        <Modal visible={visible}} onClose={() => setVisible(false)}>
            <h1>Hello</h1>
        </Modal>
      </>
    )
  }
```
